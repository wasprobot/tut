    \leavevmode\par\noindent

\Que{
    Simplify 5.3434\dots as a fraction.
}
\Ans{
    Let 
    \begin{equation}
        x = 5.3434\dots
    \end{equation}
    
    By multiplying both sides by $100$ we get\\
    \begin{equation}
        100x = 534.3434\dots
    \end{equation}
    
    Subtracting $(1)$ from $(2)$
    \begin{align*}
        100x-x &= 529\\
        99x &= 529 && \text{(next divide both sides by $99$)}\\
        x &= \frac{529}{99}\\
    \end{align*}
}
